/* 
 * Get information about the resource status of a container
 * (CPU, Memory, Net)
 */

// GET /container/:container_id/status

// Code: 404 
//   - Container not found
//   - Data not ready (container launched for less than a second)

// Code 200 - OK
// Content-Type: application/json
{
  "cpu": 44,
  "free_memory": 123456,
  "memory": 2340354,
  "net": {
    "rx": 123456,
    "tx": 123564
  }
}
